5 = number_of_philosophers and forks
800 = time_to_die / time a philosophers can survive without eating
200 = time_to_eat / time spent eating
200 = time_to_sleep / time spent sleeping
7 = numbers_of_meal / before ends

un philosophers = un thread
une fourchette = un mutex

thread = petite partie d'un programme qui s'execute en parallele
mutex = evite que deux thread utilisent la meme donnee en meme temps

pthread_t thread1 = chaque philosophe 
int id1 = numero unique de chaque philosophe 
pthread_create = cree le thread1 etc 
pthread_join = attend que le thread 1 se termine etc
pthread_mutex_init = creer et initialie le verrou \ je cree la clef
pthread_mutex_lock = verrou \ je prends ls clef
pthread_mutex_unlock = libere le verrou (sinon deadlock) \ je libere la clef
